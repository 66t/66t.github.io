var LIM=LIM||{};
LIM.WORLD=LIM.WORLD||{};
    
((_)=> {
    _.canvasWidth = 1920;  //画布大小
    _.canvasHeight = 1920;
    /////////////////////////////
    _.windowWidth=0        //画面大小
    _.windowHeight=0 
    _.scale = 1            //缩放比例
    _.app=null             //pixi应用
    
    _.tick=null            //触发
    _.clock=[0,0,0,0,0,0]  //时钟
    _.cursor=[0,0]         //鼠标
    _.canvas={x:0,y:0,w:0,h:0}  //画布位置
    
    _.showFps=false
    _.resizeTimeout=null
    
    _.fps = document.createElement("div");      //fps dom
    _.memory = document.createElement("div");   //内存 dom
    _.xy = document.createElement("div");       //光标实时位置dom
    
    //初始化
    _.init=function (){
        PIXI.utils.skipHello();
        PIXI.settings.GC_MAX_IDLE = 600;
        _.initCanvas()
        _.initFps()     
        _.updateFps()
        _.setupEventHandlers()  //注册操作
        
        
        ////测试
        const stage= new PIXI.Container();
        const sprite = new PIXI.Sprite.from('/img/back.png');
        sprite.x=0
        sprite.y=0
        stage.addChild(sprite);
        _.setStage(stage)
        ////
        
        _.resize()
    }
    _.initCanvas=function (){
        _.app = new PIXI.Application({
            width: _.canvasWidth,
            height: _.canvasHeight,
            antialias: true,
            transparent: false,
            resolution: window.devicePixelRatio,
            forceCanvas: false
        });
        
        _.app.ticker.remove(_.app.render, _.app);
        _.app.ticker.add(_.onTick, this);
        
        _.clock[2]=100
        _.clock[1]=performance.now()-100
        document.body.appendChild(_.app.view);
        _.app.view.style.position="absolute"
    }
    
    _.initFps=function (){
        document.body.appendChild(_.fps)
        _.fps.style.width="200px"
        _.fps.style.height="30px"
        _.fps.style.top="30px"
        _.fps.style.left="10px"
        _.fps.style.lineHeight="30px"
        _.fps.style.position="absolute"
        _.fps.style.backgroundColor="#000a"
        _.fps.style.color="#fff"
        _.fps.style.fontSize="18px"
        _.fps.style.fontWeight="800"
        _.fps.style.textAlign="left"
        _.fps.style.pointerEvents="none"

        document.body.appendChild(_.memory)
        _.memory.style.width="200px"
        _.memory.style.height="30px"
        _.memory.style.left="10px"
        _.memory.style.lineHeight="30px"
        _.memory.style.position="absolute"
        _.memory.style.backgroundColor="#000a"
        _.memory.style.color="#fff"
        _.memory.style.fontSize="18px"
        _.memory.style.fontWeight="800"
        _.memory.style.textAlign="left"
        _.memory.style.pointerEvents="none"

        document.body.appendChild(_.xy)
        _.xy.style.width="200px"
        _.xy.style.height="30px"
        _.xy.style.top="60px"
        _.xy.style.left="10px"
        _.xy.style.lineHeight="30px"
        _.xy.style.position="absolute"
        _.xy.style.backgroundColor="#000a"
        _.xy.style.color="#fff"
        _.xy.style.fontSize="18px"
        _.xy.style.fontWeight="800"
        _.xy.style.textAlign="left"
        _.xy.style.pointerEvents="none"
    }
    _.updateFps=function (){
        _.fps.style.display=_.showFps?"block":"none"
        _.xy.style.display=_.showFps?"block":"none"
        _.memory.style.display=_.showFps?"block":"none"
    }
    _.drawFps=function (){
        _.fps.innerHTML="fps:"+_.clock[3].toFixed(0)
        _.memory.innerHTML=(performance.memory.usedJSHeapSize/1048576).toFixed(2)+'MB'
        _.xy.innerHTML=_.cursor[0].toFixed(0)+"/"+_.cursor[1].toFixed(0)
    }
    
    //舞台
    _.setStage = function(stage) {if (_.app) _.app.stage = stage;}
    _.canRender = function() {return !!_.app.stage;}
    //设置工作
    _.setTick = function(handler) {_.tick = handler;}
    //执行工作
    _.onTick = function(deltaTime) {
        _.clock[0]=performance.now();
        
        if (_.tick) {_.tick(deltaTime);}
        if ( _.canRender()) {_.app.render();}
        const time = performance.now();
        const thisclockTime = time - _.clock[1];
        
        _.clock[2] +=(thisclockTime -  _.clock[2]) / 12;;
        _.clock[3] = 1000 /  _.clock[2];
        _.clock[4] = Math.max(0, time -  _.clock[0]);
        _.clock[1] = time;
        
        if( _.clock[5]++%15===0) _.drawFps()
       
    };
    //计算窗体大小
    _.resize =function (){
        _.windowWidth = window.innerWidth
        _.windowHeight = window.innerHeight
        let scale= Math.min(_.windowWidth/_.app.view.offsetWidth,_.windowHeight/_.app.view.offsetHeight)
        _.setScale(scale)
    }
    //改变窗体
    _.setScale = function (scale) {
        if (_.app && _.app.view) {
            _.scale=scale
            const canvasStyle = _.app.view.style;
            const canvasOffsetWidth = _.app.view.offsetWidth;
            const canvasOffsetHeight = _.app.view.offsetHeight;
            _.canvas.w=canvasOffsetWidth*scale
            _.canvas.h=canvasOffsetHeight*scale
            _.canvas.x=(_.windowWidth-_.canvas.w)/2
            _.canvas.y=(_.windowHeight-_.canvas.h)/2
            canvasStyle.transform = `scale(${scale}, ${scale})`;
            canvasStyle.left = `calc((-${canvasOffsetWidth}px * (1 - ${scale})) / 2 + (${_.windowWidth}px - ${canvasOffsetWidth}px * ${scale}) / 2)`;
            canvasStyle.top = `calc((-${canvasOffsetHeight}px * (1 - ${scale})) / 2 + (${_.windowHeight}px - ${canvasOffsetHeight}px * ${scale}) / 2)`;
        }
    }
    //光标移动
    _.cursorMove=function (event){
        let sx=event.clientX-LIM.WORLD.canvas.x
        let sy=event.clientY-LIM.WORLD.canvas.y
        _.cursor=[sx/LIM.WORLD.scale/window.devicePixelRatio,sy/LIM.WORLD.scale/window.devicePixelRatio]
    }
    //按下
    _.onKeyDown = function(event) {
        if (!event.ctrlKey && !event.altKey) {
            switch (event.keyCode) {
                case 113: // F2
                    _.showFps=!_.showFps
                    _.updateFps()
                    break
                case 115: // F4
                    if(Utils.isNwjs()) {
                        let gui = require('nw.gui');
                        let win = gui.Window.get();
                        if (window.screen.width<win.width) win.restore();    
                        else win.maximize()
                    }
                    break
                case 116: // F5
                    location.reload()
                    break;
            }
        }
    }
    //注册事件
    _.setupEventHandlers=function (){
        document.addEventListener('mousemove',_.cursorMove.bind(this));
        document.addEventListener("keydown", _.onKeyDown.bind(this));
        window.addEventListener('resize', function() {
            clearTimeout(_.resizeTimeout);
            _.resizeTimeout = setTimeout( ()=>{_.resize();}, 100);
        });
    }
    
})(LIM.WORLD);
